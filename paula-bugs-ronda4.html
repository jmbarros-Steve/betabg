const expected = createHmac('sha256', shopifySecret)
  .update(`${h}.${p}`).digest('base64url');

▶ #104 high OAuth state nunca consumido — replay attack + table leak oauth/meta-oauth-callback.ts:31-38

El state de OAuth se valida pero nunca se borra de oauth_states. El token es reutilizable para replay attacks. Además no se chequea expires_at. Comparar con Shopify OAuth que sí borra el state después de usarlo.

Codigo problematico:
const { data: stateCheck } = await supabase
  .from('oauth_states').select('id').eq('state', state).maybeSingle();
if (!stateCheck) return 403;
// No delete! State can be reused
Fix sugerido:
// Validate + consume (one-time use)
const { data: stateCheck } = await supabase
  .from('oauth_states').select('id, expires_at')
  .eq('state', state).maybeSingle();

if (stateCheck.expires_at && new Date(stateCheck.expires_at) < new Date())
  return 403;
await supabase.from('oauth_states').delete().eq('id', stateCheck.id);

▶ #109 medium Open redirect sin validacion en WebForm redirect_url pages/WebForm.tsx:70-73

Después del submit, window.location.href = data.redirect_url sin validar. Un atacante puede setear redirect_url a javascript:alert() o un dominio de phishing. La página es pública (sin auth).

Codigo problematico:
if (data.redirect_url) {
  setTimeout(() => {
    window.location.href = data.redirect_url; // no validation
  }, 2000);
}
Fix sugerido:
if (data.redirect_url) {
  try {
    const parsed = new URL(data.redirect_url, window.location.origin);
    if (parsed.protocol === 'https:' || parsed.protocol === 'http:') {
      setTimeout(() => { window.location.href = parsed.href; }, 2000);
    }
  } catch { /* block invalid URL */ }
}
